# 单向链表

## 洛谷B3631

```c++
#include<iostream>
using namespace std;
int main(){
    int q,a[1000005],op,next;
    cin>>q;
    while(q--){
        cin>>op;
        if(op==1){
            int x,y;
            cin>>x>>y;
            next=a[x];
            a[x]=y;
            a[y]=next;
        }
        else if(op==2){
            int x;
            cin>>x;
            cout<<a[x]<<endl;
        }
        else{
            int x;
            cin>>x;
            next=a[x];
            a[x]=a[next];
        }
    }
    return 0;
}
```
