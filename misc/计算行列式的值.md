# 计算行列式的值

```c++
#include<iostream>
#include<vector>

using namespace std;
#define ll long long

#define endl '\n'
const int N = 2e5 + 5, mod = 1e9 + 7;
int t;

int det(int a[N][N], int n);

int minor(int a[N][N], int i, int n);

int det(int a[N][N], int n) {//计算行列式的值
    if (n == 1)//一阶行列式就是第一个
        return a[0][0];
    int sum = 0, m;
    for (int i = 0; i < n; i++) {//按第一行展开
        m = minor(a, i, n);//计算余子式
        sum += pow(-1, i + 2) * a[0][i] * m;//展开公式
    }
    return sum;
}

int minor(int a[N][N], int i, int n) {//构造余子式
    int a2[N][N];
    for (int j = 0; j < n - 1; j++)//行和列都比之前少一个
        for (int k = 0; k < n - 1; k++) {
            if (k < i)
                a2[j][k] = a[j + 1][k];
            else// k>=i 
                a2[j][k] = a[j + 1][k + 1];
        }
    return det(a2, n - 1);//新的余子式在拿去det计算行列式的值
}

void solve() {
    int n, arr[N][N];
    cin >> n;//行列式阶数
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cin >> arr[i][j];
    cout << det(arr, n);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    t = 1;
    //cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
