int a[100005], tmp[1001];
//归并排序(分治)
void merge(int l, int mid, int r) {
	int i = l, j = mid + 1, k = l;
	while (i <= mid && j <= r) {
		if (a[i] <= a[j])//谁小先放谁
			tmp[k++] = a[i++];
		else
			tmp[k++] = a[j++];
	}
	while (i <= mid)//上面操作可能有一边没放完,接着放
		tmp[k++] = a[i++];
	while (j <= r)
		tmp[k++] = a[j++];
	for (int i = l; i <= r; i++)//把排列好的数组赋给原数组
		a[i] = tmp[i];
}
void mergesort(int l, int r) {
	if (l >= r)
		return;
	int mid = l + r >> 1;
	mergesort(1, mid);
	mergesort(mid + 1, r);
	merge(l, mid, r);//分完组排列
}
int  main()
{
	ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	mergesort(1, n);
	for (int i = 1; i <= n; i++)
		cout << a[i] << ' ';
	return 0;
}
