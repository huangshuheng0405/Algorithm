# 编辑距离

```c++
int editdist(string &a, string &b) {
    int m = a.size();
    int n = b.size();
    vector<vector<int>> dp(m, vector<int>(n + 1));
    for (int i = 1; i <= m; i++) {
        dp[i][0] = i;
    }
    for (int j = 1; j <= n; j++) {
        dp[0][j] = j;
    }
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (a[i - 1] == b[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1];
            } else {
                dp[i][j] = std::min({dp[i - 1][j - 1], dp[i - 1][j], dp[j][i - 1]}) + 1;
            }
        }
    }
    
    return dp[m][n];
}
```
